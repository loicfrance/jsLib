mix(superclass, mixins) → class

creates a mix of a superclass and several mixins to make a class extend a class and implements mixins.

Example 

class A {
	   constructor(x) {
	       this.x = x;
	   }
	   hello() {
	   	   alert('hello ' + this.x);
	   }
};
var B = {
     hello: function() {
	       		alert('Hi ' + this.x + '!');
	       }
	   }
};
class C extends utils.tools.mix(A, B) {
    constructor(x) {
		   super(x);
	   }
	   hello() {
	       super.hello();
	       alert('How are you ?');
	   }
}

var d = new D('John');
d.hello(); // alert('Hi John !'); alert('how are you ?');
